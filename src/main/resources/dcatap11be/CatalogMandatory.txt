# Missing mandatory properties

PREFIX dcat:    <http://www.w3.org/ns/dcat#>
PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?catalog ?property
WHERE {
    ?catalog a dcat:Catalog
    FILTER NOT EXISTS { 
        ?catalog ?property ?value
    }
    VALUES ?property {
        dcterms:title
        dcterms:description
        dcterms:publisher
        dcat:dataset
    } 
}
