# Distribution missing recommended properties

PREFIX dcat:    <http://www.w3.org/ns/dcat#>
PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?distribution ?property
WHERE {
    ?distribution a dcat:Distribution
    FILTER NOT EXISTS { 
        ?distribution ?property ?value
    }
    VALUES ?property {
        dcterms:title
        dcterms:description
        dcterms:license
        dcterms:language
    } 
}
