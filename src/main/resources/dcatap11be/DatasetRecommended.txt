# Dataset missing recommended properties

PREFIX dcat:    <http://www.w3.org/ns/dcat#>
PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?dataset ?property
WHERE {
    ?dataset a dcat:Dataset
    FILTER NOT EXISTS { 
        ?dataset ?property ?value
    }
    VALUES ?property {
        dcat:theme
        dcat:contactPoint
        dcterms:publisher
    }
}
